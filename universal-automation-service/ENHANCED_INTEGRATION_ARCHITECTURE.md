"""Enhanced EventRelay Client with MCP Integration"""
import httpx
from typing import Dict, Any
from clients.mcp_orchestrator_client import MCPOrchestratorClient

class EnhancedEventRelayClient:
    def __init__(self, base_url: str = "http://localhost:8000"):
        self.base_url = base_url
        self.client = httpx.AsyncClient()
        self.mcp = MCPOrchestratorClient()

    async def process_video_with_research(self, video_url: str) -> Dict[str, Any]:
        # Stage 1: Get transcript from EventRelay
        transcript_result = await self.client.post(
            f"{self.base_url}/api/v1/transcript-action",
            json={"video_url": video_url, "language": "en"}
        )
        transcript_data = transcript_result.json()

        # Stage 2: Store in Supabase
        await self.mcp.store_in_supabase("video_transcripts", {
            "video_url": video_url,
            "transcript": transcript_data.get("transcript"),
            "events": transcript_data.get("events")
        })

        # Stage 3: Research mentioned technologies with Perplexity
        topics = transcript_data.get("key_topics", [])
        research_results = {}
        for topic in topics[:5]:  # Top 5 topics
            research = await self.mcp.research_with_perplexity(
                f"Best practices and documentation for {topic}"
            )
            research_results[topic] = research

        # Stage 4: Track analytics
        await self.mcp.track_analytics("video_processed", {
            "video_url": video_url,
            "topics_count": len(topics),
            "transcript_length": len(transcript_data.get("transcript", ""))
        })

        return {
            "transcript": transcript_data,
            "research": research_results,
            "stored_in_db": True
        }
